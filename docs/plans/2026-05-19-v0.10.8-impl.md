```ts
// At top of file, add import:
import lockfile from "proper-lockfile";
import { dirname } from "node:path";
import { mkdirSync } from "node:fs";

// Existing renderEventBlock(...) stays unchanged.

// Add new pure helper alongside `renderEventBlock`:
function renderJsonlLine(event: BrainLogEntry): string {
  // The JSONL row is a one-to-one projection of the markdown body:
  // every bullet becomes a key, array bullets become JSON arrays.
  // `BrainLogEntry.body` already has that shape, so the projection is
  // direct - no per-kind branching, which keeps the writer kind-agnostic.
  const payload: Record<string, string | ReadonlyArray<string>> = {};
  for (const [key, value] of Object.entries(event.body)) {
    payload[key] = value;
  }
  return JSON.stringify({
    ts: event.timestamp,
    kind: event.eventType,
    payload,
  });
}

// Existing `appendLogEvent` becomes an async-free function that uses a
// synchronous proper-lockfile lock - the public signature stays the same
// (synchronous), callers do not change.

export function appendLogEvent(
  vault: string,
  event: BrainLogEntry,
): AppendLogEventResult {
  if (!EVENT_KIND_VALUES.has(event.eventType)) {
    throw new Error(
      `appendLogEvent: unknown event kind '${event.eventType}' - must be one of ${Object.values(
        BRAIN_LOG_EVENT_KIND,
      ).join(", ")}`,
    );
  }

  const ts = parseIsoUtc(event.timestamp);
  const mdPath = logPath(vault, ts.date);
  const jsonlPath = logJsonlPath(vault, ts.date);

  // proper-lockfile requires the lock target to exist. Lock the log
  // directory itself, created up-front, the same scope used by
  // `src/core/event-log.ts:appendEvent`. The sync API forbids the
  // built-in `retries` option, so the call is wrapped in a small
  // ELOCKED-only retry helper (see `acquireLogLock` in the shipped
  // `log.ts`).
  const logDir = brainDirs(vault).log;
  mkdirSync(logDir, { recursive: true });

  const release = acquireLogLock(logDir);

  try {
    // ---- JSONL sidecar (machine surface, primary) ---------------------
    // Write JSONL first: it is the source of truth for `readLogDay`.
    // If the markdown write below fails, the human Obsidian view lags
    // by one event, but discipline-report / future machine consumers
    // keep returning the correct counts.
    const existingJsonl = existsSync(jsonlPath) ? readFileSync(jsonlPath, "utf8") : "";
    const line = renderJsonlLine(event);
    const nextJsonl =
      existingJsonl === ""
        ? line + "\n"
        : `${existingJsonl.replace(/\s+$/, "")}\n${line}\n`;
    atomicWriteFileSync(jsonlPath, nextJsonl);

    // ---- Markdown (human surface) -------------------------------------
    const existingMd = existsSync(mdPath) ? readFileSync(mdPath, "utf8") : "";
    const block = renderEventBlock(event, ts.hms);
    let nextMd: string;
    if (existingMd === "") {
      nextMd = renderFileHeader(ts.date) + "\n" + block;
    } else {
      nextMd = `${existingMd.replace(/\s+$/, "")}\n\n${block}`;
    }
    if (!nextMd.endsWith("\n")) nextMd += "\n";
    atomicWriteFileSync(mdPath, nextMd);

    return { logPath: mdPath };
  } finally {
    release();
  }
}
```

Note: `proper-lockfile` exposes `lockSync` (mirror of `lock`) that returns a release function; we use it to keep `appendLogEvent` synchronous (the existing public contract). The sync flavour does not support the `retries` option, so the shipped implementation wraps `lockSync` in a small ELOCKED-only retry loop (`acquireLogLock`) instead of passing `retries` directly.

```ts
/**
 * Brain log reader that prefers the v0.10.8 JSONL sidecar and falls
 * back to parsing the markdown on days without one. All machine
 * consumers (discipline-report today, future tooling tomorrow) read
 * through this entry point so they do not duplicate the fallback
 * logic.
 *
 * The reader returns the same `BrainLogEntry[]` shape that
 * `parseLogDay` returns, plus a `source` discriminator the caller can
 * surface in diagnostics. Malformed JSONL lines do not abort the
 * read; they surface as `warnings` so a partially-edited file is
 * still useful.
 */

import { existsSync, readFileSync } from "node:fs";

import { logPath, validateIsoDate } from "./paths.ts";
import {
  parseLogDay,
  type BrainLogEntry,
  type BrainLogParseWarning,
} from "./log.ts";
import {
  BRAIN_LOG_EVENT_KIND,
  type BrainLogEventKind,
} from "./types.ts";

export interface ReadLogDayResult {
  readonly entries: ReadonlyArray<BrainLogEntry>;
  readonly source: "jsonl" | "markdown-fallback";
  readonly warnings: ReadonlyArray<BrainLogParseWarning>;
}

const EVENT_KIND_VALUES: ReadonlySet<string> = new Set(
  Object.values(BRAIN_LOG_EVENT_KIND),
);

export function readLogDay(vault: string, date: string): ReadLogDayResult {
  const validDate = validateIsoDate(date);
  const mdPath = logPath(vault, validDate);
  const jsonlPath = mdPath.replace(/\.md$/, ".jsonl");

  if (existsSync(jsonlPath)) {
    return readJsonl(jsonlPath);
  }
  if (existsSync(mdPath)) {
    const parsed = parseLogDay(vault, validDate);
    return {
      entries: parsed.entries,
      source: "markdown-fallback",
      warnings: parsed.warnings,
    };
  }
  return { entries: [], source: "jsonl", warnings: [] };
}

function readJsonl(path: string): ReadLogDayResult {
  let text: string;
  try {
    text = readFileSync(path, "utf8");
  } catch (err) {
    // The caller already verified `existsSync(path)`; a read failure
    // here is anomalous (rotation race, permission flip, transient
    // fs error). Surface it as a warning so discipline-report and
    // doctor can flag the day instead of silently returning zero.
    const message = (err as NodeJS.ErrnoException).message ?? String(err);
    return {
      entries: [],
      source: "jsonl",
      warnings: [{ path, lineNumber: 0, message: `failed to read JSONL file: ${message}` }],
    };
  }

  const entries: BrainLogEntry[] = [];
  const warnings: BrainLogParseWarning[] = [];
  const lines = text.split(/\r?\n/);
  for (let i = 0; i < lines.length; i++) {
    const line = lines[i]!;
    if (line.trim() === "") continue;
    let parsed: unknown;
    try {
      parsed = JSON.parse(line);
    } catch {
      warnings.push({
        path,
        lineNumber: i + 1,
        message: `malformed JSONL line: ${line.slice(0, 80)}`,
      });
      continue;
    }
    const entry = coerceEntry(parsed, path, i + 1, warnings);
    if (entry !== null) entries.push(entry);
  }
  return { entries, source: "jsonl", warnings };
}

function coerceEntry(
  raw: unknown,
  path: string,
  lineNumber: number,
  warnings: BrainLogParseWarning[],
): BrainLogEntry | null {
  if (raw === null || typeof raw !== "object") {
    warnings.push({ path, lineNumber, message: "JSONL row is not an object" });
    return null;
  }
  const obj = raw as Record<string, unknown>;
  const ts = obj["ts"];
  const kind = obj["kind"];
  const payload = obj["payload"];
  if (typeof ts !== "string" || typeof kind !== "string") {
    warnings.push({
      path,
      lineNumber,
      message: "JSONL row missing ts/kind",
    });
    return null;
  }
  if (!EVENT_KIND_VALUES.has(kind)) {
    warnings.push({ path, lineNumber, message: `unknown event kind: ${kind}` });
    return null;
  }
  if (payload === null || typeof payload !== "object" || Array.isArray(payload)) {
    warnings.push({ path, lineNumber, message: "JSONL row missing payload object" });
    return null;
  }
  const body: Record<string, string | ReadonlyArray<string>> = {};
  for (const [k, v] of Object.entries(payload as Record<string, unknown>)) {
    if (typeof v === "string") {
      body[k] = v;
    } else if (Array.isArray(v) && v.every((x) => typeof x === "string")) {
      body[k] = v as string[];
    } else {
      warnings.push({
        path,
        lineNumber,
        message: `JSONL payload key '${k}' has unsupported value type`,
      });
    }
  }
  return {
    timestamp: ts,
    eventType: kind as BrainLogEventKind,
    body: Object.freeze(body),
  };
}
```
